📦 Установка 🪟 Окно 📑 Вкладки 📂 Элементы 📄 Пример ❌ Закрытие

📦 Установка

lualocal UI = loadstring(game:HttpGet("https://pastebin.com/raw/ТВОЙ_КОД"))()

🪟 Создание окна

lualocal win = UI:CreateWindow("Название", {
    Width = 520,
    Height = 600,
    X = 0.5,
    Y = 0.5
})

📑 Вкладки

lualocal tab = win:CreateTab("Название вкладки")

📂 Элементы

▸ Секция

luatab:CreateSection("Текст")

▸ Кнопка

luatab:CreateButton("Текст", function()
    print("Нажата!")
end)

▸ Тоггл

luatab:CreateToggle("Название", false, function(state)
    print(state)
end)

▸ Слайдер

luatab:CreateSlider("Название", 16, 300, 16, "", function(v)
    print(v)
end)

▸ Дропдаун

luatab:CreateDropdown("Название", {"А", "Б"}, "А", function(v)
    print(v)
end)

📄 Полный пример

lualocal UI = loadstring(game:HttpGet("https://pastebin.com/raw/ТВОЙ_КОД"))()
local win = UI:CreateWindow("Моё меню")
local main = win:CreateTab("Главная")
main:CreateSection("Управление")
main:CreateButton("Привет", function() print("Привет!") end)
main:CreateToggle("Ноклип", false, function(s) print(s) end)
main:CreateSlider("Скорость", 16, 300, 16, "", function(v)
    game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = v
end)
main:CreateDropdown("Предмет", {"Бита", "Молоток"}, "Бита", print)
